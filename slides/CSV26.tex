\begin{frame}{Verification of smart contracts}
  \begin{itemize}
  \item the importance of software verification is obvious for our community
  \item smart contracts deal with money: still more relevant
  \item many tools are based on static analysis
    (Slither, Mythril, Solhint, Securify, Rattle, LiSA, \ldots)
  \end{itemize}

  \begin{center}
    \textbf{So why testing?}
  \end{center}

  Testing has limits but also its own advantages:

  \begin{itemize}
  \item stable results, independent from the version of the testing framework
  \item (hopefully) quick results
  \item focus on more relevant cornercases
  \item focus on the intention of the programmer, not on abstract correctness
  \item mature testing frameworks
  \item simple integration with IDEs and continuous integration suites
  \end{itemize}
\end{frame}

\begin{frame}\frametitle{Origin of this work}
  This work originated during the development of the Hotmoka layer for the execution of
  smart contracts in Java:

  \begin{center}
    {\color{red} Olivieri, Spoto and Tagliaferro:
    \emph{An application layer with protocol-based Java smart contract verification},
    Frontiers in Computer Science, 7, 2025}
  \end{center}

  \begin{itemize}
  \item Hotmoka applies static analysis at the protocol-level: a filter to reject
  inconsistent smart contracts

  \item Hotmoka allows testing during development: a check of the
  smart contracts against the intention of the programmer
  \end{itemize}

\end{frame}

\begin{frame}[fragile]\frametitle{A smart contract for a Ponzi scheme}

  \begin{lstlisting}[language=Takamaka]
public class Ponzi extends Contract {
 private final StorageList<PayableContract> investors = new StorageLinkedList<>();

 public @FromContract(PayableContract.class) Ponzi() {
  investors.add((PayableContract) caller());
 }

 public @Payable @FromContract(PayableContract.class) void invest(long amount) {
  require(amount >= 1000, "you must invest at least 1000");
  var split = amount / investors.size();
  investors.forEach(investor -> investor.receive(split));
  investors.add((PayableContract) caller());
 }
}
  \end{lstlisting}

\end{frame}

\begin{frame}[fragile]\frametitle{A JUnit test class for the Ponzi scheme contract 2/3}

  \begin{lstlisting}[language=Java]
 @Test void investmentSplitsCorrectly() {
  // send to Hotmoka a request to let account(0) create
  // the contract and become the first investor
  var contract = callConstructor(privateKey(0), account(0), GAS_LIMIT, GAS_PRICE, jar(), PONZI_CONSTRUCTOR);
  // take note of the balance of account(0) now
  var initial = getBalanceOf(account(0));
  // account(1) invests in the contract
  var first = 10_000_000;
  callMethod(privateKey(1), account(1), GAS_LIMIT, GAS_PRICE, jar(), INVEST, contract, first);
  // the first investment went to account(0)
  assertEquals(initial + first, getBalanceOf(account(0)));
  // account(2) invests in the contract
  var second = 2_000_000;
  callMethod(privateKey(2), account(2), GAS_LIMIT, GAS_PRICE, jar(), INVEST, contract, second);
  // half of the second investment went to account(0)
  assertEquals(initial + first + second/2, getBalanceOf(account(0)));
 }
  \end{lstlisting}

\end{frame}

\begin{frame}\frametitle{The API of a Hotmoka node}

  JUnit tests of Hotmoka smart contracts work against the abstract interface of a node,
  consisting of requests to:

  \begin{itemize}
  \item install a jar (paid)
  \item create an object (paid)
  \item call a method of an object or of a class (paid)
  \item run a side-effect-free method of an object or of a class (free)
  \end{itemize}

  \medskip

  Requests can receive parameters and yield a result:
  \begin{itemize}
  \item primitive values (\texttt{int}, \texttt{float}, \texttt{boolean}, \ldots)
  \item machine-independent references (addresses inside the database of states)
  \end{itemize}
\end{frame}

\begin{frame}\frametitle{Different implementations of the same node API interface}

  The test suite's fixture instantiates a test \<node> to run all tests against:

  \begin{itemize}
  \item a single proof of stake local node: \<node = new TendermintNode(config)>
  \item a single proof of space local node: \<node = new MokamintNode(config)>
  \item a single moked blockchain node: \<node = new DiskNode(config)>
  \item a network of $n$ proof of stake nodes: \<new TendermintNode(config$_1$)>,\ldots,
    \<new TendermintNode(config$_{n-1}$)>, \<node = new TendermintNode(config$_n$)>
  \item a network of $n$ proof of space nodes: \<new MokamintNode(config$_1$)>,\ldots,
    \<new MokamintNode(config$_{n-1}$)>, \<node = new MokamintNode(config$_n$)>
  \end{itemize}
\end{frame}

\begin{frame}\frametitle{The same node API interface also for remote nodes}

  Just use the {\color{red}adaptor design pattern}:

  \begin{itemize}
  \item a remote network of $n$ proof of space nodes:
    \begin{itemize}
    \item {\color{blue}in the remote machine}:
      \<new MokamintNode(config$_1$)>,\ldots,
      \<new MokamintNode(config$_{n-1}$)>, \<temp = new MokamintNode(config$_n$)>
      then adapt the local node \<temp> to a web service of its API listening at port 8001:
      \[
      \<new NodeService(temp, 8001)>
      \]
    \item {\color{blue}in the local (testing) machine}, adapt back the web service to a node:
      \[
      \<node = new RemoteNode("ws://remote-machine.com:8001")>
      \]
      or
      \[
      \<node = new RemoteNode("ws://localhost:8001")>
      \]
      if remote and local machine coincide.
    \end{itemize}
  \end{itemize}
  
\end{frame}

\begin{frame}\frametitle{Conclusion}
  \begin{itemize}
  \item tests are written only once but can be run against different instantiations
    of the same node API interface
    \begin{itemize}
    \item good for testing different consensus engines
    \end{itemize}
  \item testing against real consensus engines is too slow for
    frequent continuos integration or as non-regression test 
  \item a moked blockchain node that still implements a database allows one to test smart contracts
    and the Hotmoka execution layer in a few minutes
    \begin{itemize}
    \item good for frequent continuos integration and as non-regression test 
    \end{itemize}
  \item adapting web services to the node interface allows one to test against a node
    on a remote machine and to test the blockchain networking layer as well
  \item limit: consensus engines with probabilistic finality make testing hard with more than one node
    because of history changes
  \item limit: we cannot write tests depending on timings since they depend on the block creation rate
  \end{itemize}
\end{frame}
